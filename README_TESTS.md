"""
Модульные тесты для игры «Судоку» (Вариант 7).
Авторы: Некрасова В., [Фамилия партнёра]
Запуск: pytest tests/test_sudoku.py -v

class TestIsValid:
    """Тесты функции валидации хода."""

    def setup_method(self):
        """Инициализация тестового поля."""
        self.board = [
            [5, 3, 0, 0, 7, 0, 0, 0, 0],
            [6, 0, 0, 1, 9, 5, 0, 0, 0],
            [0, 9, 8, 0, 0, 0, 0, 6, 0],
            [8, 0, 0, 0, 6, 0, 0, 0, 3],
            [4, 0, 0, 8, 0, 3, 0, 0, 1],
            [7, 0, 0, 0, 2, 0, 0, 0, 6],
            [0, 6, 0, 0, 0, 0, 2, 8, 0],
            [0, 0, 0, 4, 1, 9, 0, 0, 5],
            [0, 0, 0, 0, 8, 0, 0, 7, 9]
        ]

    def test_valid_placement(self):
        """TC-Unit-001: Допустимая цифра в пустую клетку."""
        assert is_valid(self.board, 0, 2, 4) is True

    def test_invalid_row(self):
        """TC-Unit-002: Цифра уже есть в строке."""
        assert is_valid(self.board, 0, 2, 5) is False  # 5 уже в строке 0

    def test_invalid_column(self):
        """TC-Unit-003: Цифра уже есть в столбце."""
        assert is_valid(self.board, 2, 0, 6) is False  # 6 уже в столбце 0

    def test_invalid_box(self):
        """TC-Unit-004: Цифра уже есть в квадрате 3×3."""
        assert is_valid(self.board, 2, 2, 8) is False  # 8 уже в верхнем левом блоке

    def test_zero_cell_accepted(self):
        """TC-Unit-005: Валидация работает только для пустых клеток (0)."""
        # Попытка «заменить» исходную цифру — логика должна отклонить на уровне main.py,
        # но is_valid технически вернёт True, если цифра допустима
        assert is_valid(self.board, 0, 0, 5) is True  # 5 уже стоит, но формально допустима


class TestGenerateBoard:
    """Тесты генерации судоку."""

    @pytest.mark.parametrize("difficulty", ["easy", "medium", "hard"])
    def test_generate_returns_valid_structure(self, difficulty):
        """TC-Unit-006: Генерация возвращает два поля 9×9."""
        puzzle, solution = generate_board(difficulty)
        assert len(puzzle) == 9 and all(len(row) == 9 for row in puzzle)
        assert len(solution) == 9 and all(len(row) == 9 for row in solution)

    def test_solution_is_complete(self):
        """TC-Unit-007: Решение не содержит нулей."""
        _, solution = generate_board("easy")
        zeros = sum(cell for row in solution for cell in row if cell == 0)
        assert zeros == 0

    def test_puzzle_has_empty_cells(self):
        """TC-Unit-008: Загадка содержит пустые клетки."""
        puzzle, _ = generate_board("easy")
        empty = sum(1 for row in puzzle for cell in row if cell == 0)
        assert empty > 0  # Хотя бы одна пустая клетка


class TestSolve:
    """Тесты решателя (backtracking)."""

    def test_solve_completes_valid_puzzle(self):
        """TC-Unit-009: Решатель завершает валидное судоку."""
        puzzle, solution = generate_board("easy")
        test_board = [row[:] for row in puzzle]
        result = solve(test_board)
        assert result is True
        assert test_board == solution  # После решения должно совпадать с эталоном


class TestBoundaryValues:
    """Параметризованные тесты граничных значений."""

    @pytest.mark.parametrize("row, col, num, expected", [
        # Валидные границы
        (0, 0, 1, True),   # верхний левый угол
        (8, 8, 9, True),   # нижний правый угол
        # Невалидные: повтор в строке
        (0, 2, 5, False),  # 5 уже в строке 0
        # Невалидные: повтор в столбце
        (2, 0, 6, False),  # 6 уже в столбце 0
        # Невалидные: повтор в блоке
        (2, 2, 8, False),  # 8 уже в блоке (0-2, 0-2)
    ])
    def test_is_valid_boundaries(self, row, col, num, expected):
        """TC-Unit-010: Валидация на граничных случаях."""
        board = [
            [5, 3, 0, 0, 7, 0, 0, 0, 0],
            [6, 0, 0, 1, 9, 5, 0, 0, 0],
            [0, 9, 8, 0, 0, 0, 0, 6, 0],
            [8, 0, 0, 0, 6, 0, 0, 0, 3],
            [4, 0, 0, 8, 0, 3, 0, 0, 1],
            [7, 0, 0, 0, 2, 0, 0, 0, 6],
            [0, 6, 0, 0, 0, 0, 2, 8, 0],
            [0, 0, 0, 4, 1, 9, 0, 0, 5],
            [0, 0, 0, 0, 8, 0, 0, 7, 9]
        ]
        assert is_valid(board, row, col, num) == expected
